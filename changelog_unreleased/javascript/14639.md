#### Fix incorrect formatting of member chains where last call has a single template literal argument (#14639 by @solarized-fox)

<!-- prettier-ignore -->
```jsx
// Input
foo("long" + "set").bar("of" + "arguments").baz("will" + "break")
.qux(`
    template literal
`)

insertRule(`*, *:before, *:after {
  box-sizing: inherit;
}`);

// Prettier stable
foo("long" + "set")
  .bar("of" + "arguments")
  .baz("will" + "break").qux(`
    template literal
`);

insertRule(`*, *:before, *:after {
   box-sizing: inherit;
}`);

// Prettier main
foo("long" + "set")
  .bar("of" + "arguments")
  .baz("will" + "break")
  .qux(`
    template literal
`);

insertRule(
   `*, *:before, *:after {
   box-sizing: inherit;
 }`);
```
