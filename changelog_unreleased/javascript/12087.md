#### Constant line breaks for Conditional operators (#12087 by @islandryu)

```js
//--print-width=25
// Input
const a = foo == null ? bar: baz;

const b = !fooooooooo ? bar: baz;

const x = {
  prop1: foo == null ? bar : baz,
  prop1: !fooooooooo ? bar : baz,
};

// Prettier stable
const a =
  foo == null
    ? bar
    : baz;

const b = !fooooooooo
  ? bar
  : baz;

const x = {
  prop1:
    foo == null
      ? bar
      : baz,
  prop1: !fooooooooo
    ? bar
    : baz,
};


// Prettier main
const a =
  foo == null
    ? bar
    : baz;

const b =
  !fooooooooo
    ? bar
    : baz;

const x = {
  prop1:
    foo == null
      ? bar
      : baz,
  prop1:
    !fooooooooo
      ? bar
      : baz,
};
```
