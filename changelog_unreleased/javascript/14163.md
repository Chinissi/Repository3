#### Fix cursor tracking inside JSX Text (#14163 by @fisker)

<!-- prettier-ignore -->
```console
// Prettier stable
require("prettier").formatWithCursor(
  ["<>a", "  <div>hi</div>", "</>"].join("\n"),
  { cursorOffset: 3, parser: "babel" }
).cursorOffset;

// -> 2

// Prettier main
require("prettier").formatWithCursor(
  ["<>a", "  <div>hi</div>", "</>"].join("\n"),
  { cursorOffset: 3, parser: "babel" }
).cursorOffset;

// -> 6
```
