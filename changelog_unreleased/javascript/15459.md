#### Allow getters and setters to be printed on one line (#15459 by @rattrayalex)

<!-- prettier-ignore -->
```ts
// Input
class Foo {
  get foo() { return this._foo; }
  set foo(foo) { this._foo = foo; }
}

// Prettier stable
class Foo {
  get foo() {
    return this._foo;
  }
  set foo(foo) {
    this._foo = foo;
  }
}

// Prettier main
class Foo {
  get foo() { return this._foo; }
  set foo(foo) { this._foo = foo; }
}
```
