#### Ignore errors for sloppy mode syntax (#11750 by @fisker, #11778 by @sosukesuzuki)

<!-- prettier-ignore -->

// Input
function foo() { var bar = 1; delete bar; }

// Prettier stable
SyntaxError: Deleting local variable in strict mode. (1:31)
> 1 | function foo() { var bar = 1; delete bar; }
    |                               ^

// Prettier main
function foo() {
  var bar = 1;
  delete bar;
}
