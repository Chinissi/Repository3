#### fix formatting for typescript Enum with computed members (#12930 by @HosokawaR)

<!-- prettier-ignore -->
```tsx
// Input
enum A {
  [i++],
}

const bar = "bar"
enum B {
  [bar] = 2,
}

const foo = () => "foo";
enum C {
  [foo()] = 2,
}

// Prettier stable
enum A {
  i++,
}

const bar = "bar";
enum B {
  bar = 2,
}

const foo = () => "foo";
enum C {
  foo() = 2,
}

// Prettier main
enum A {
  [i++],
}

const bar = "bar";
enum B {
  [bar] = 2,
}

const foo = () => "foo";
enum C {
  [foo()] = 2,
}
```
