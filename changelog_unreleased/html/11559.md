#### Respect single-quote option in HTML attributes (#11559 by @dcyriller)

<!-- prettier-ignore -->
```sh
--single-quote
```

<!-- prettier-ignore -->
```html
<!-- Input -->
<div class='foo' title='My title'></div>
<div class="foo" title="My title"></div>
<div class='foo "bar"' title='My "title"'></div>
<div class="foo 'bar'" title="My 'title'"></div>

<!-- Prettier stable -->
<div class="foo" title="My title"></div>
<div class="foo" title="My title"></div>
<div class="foo &quot;bar&quot;" title='My "title"'></div>
<div class="foo 'bar'" title="My 'title'"></div>

<!-- Prettier main -->
 <div class='foo' title='My title'></div>
 <div class='foo' title='My title'></div>
 <div class='foo "bar"' title='My "title"'></div>
 <div class="foo 'bar'" title="My 'title'"></div>
```
