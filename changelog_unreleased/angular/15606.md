#### Support Angular control flow (#15606 by @DingWeizhe)

@if / @else

<!-- prettier-ignore -->
```html
<!-- Input -->
@if (isChecked()) {
<span>Checked</span>
}
@else {
<span>Not checked</span>
}

<!-- Prettier stable -->
@if (isChecked()) {
  <span>Checked</span>
} @else {
  <span>Not checked</span>
}
```

@for / @empty

<!-- prettier-ignore -->
```html
<!-- Input -->
<ul>
@for (item of emptyCollection; track item.id;) {
<li><strong>{{item.name}}</strong></li>
}
@empty {
<span>The collection is empty</span>
}
</ul>

<!-- Prettier stable -->
<ul>
  @for (item of emptyCollection; track item.id) {
    <li>
      <strong>{{ item.name }}</strong>
    </li>
  } @empty {
    <span>The collection is empty</span>
  }
</ul>
```
